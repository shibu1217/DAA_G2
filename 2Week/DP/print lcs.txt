//topdown lcs
#include<bits/stdc++.h>

using namespace std;


int static t[1001][1001];


int lcs(string x,string y,int n,int m)
{

 int i,j;   

 
for(i=0;i<=n;i++)
{
 
   for(j=0;j<=m;j++)
    {
   
  if(i==0||j==0)
          
t[i][j]==0;
    }
        
          
 
}



for(i=1;i<=n;i++)
{
  
  for(j=1;j<=m;j++)
    {
        

if(x[i-1]==y[j-1])
 t[i][j]=1+t[i-1][j-1];


else
t[i][j]=max(t[i][j-1],t[i-1][j]);
    
}
}
string s=" ";
while(i>0 &&j>0)
{
    if(x[i-1]==y[j-1])
      {
          s.push_back(x[i-1]);
          i--;
          j--;
      }
      else 
      {
          if(t[i][j-1]>t[i-1][j])
             j--;
          else
             i--;
      }
}
reverse(s.begin(),s.end());
//for(i=0;i<n+1;i++)
cout<<s;
    

}


int main()
{
 
  // memset(t,0, sizeof(t));
  
 string x={"acbcf"};
  
  

 string y={"abcdaf"};
  
  int n,m,ans;
    
   
 n=x.length();
   
 m=y.length();
   
lcs(x,y,n,m);
  // cout<<ans;
 
  
  return 0;

}
