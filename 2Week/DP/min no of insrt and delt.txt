//topdown lcs
#include<bits/stdc++.h>

using namespace std;


int static t[1001][1001];


int lcs(string x,string y,int n,int m)
{

 int i,j;   

 
for(i=0;i<=n;i++)
{
 
   for(j=0;j<=m;j++)
    {
   
  if(i==0||j==0)
          
t[i][j]==0;
    }
        
          
 
}



for(i=1;i<=n;i++)
{
  
  for(j=1;j<=m;j++)
    {
        

if(x[i-1]==y[j-1])
 t[i][j]=1+t[i-1][j-1];


else
t[i][j]=max(t[i][j-1],t[i-1][j]);
    
}
}
return t[n][m];
}
void iad(string x,string y,int n,int m)
{
    int len=lcs(x,y,n,m);
    cout<<"\nNo of deletion:"<<n-len;
    cout<<"\n No of insertion:"<<m-len;
}

int main()

  int n,m,ans;
    
   
 n=x.length();

 // cout<<n+m-ans;
